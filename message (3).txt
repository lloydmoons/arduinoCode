#include <OneWire.h>
#include <DallasTemperature.h>
#include <DHT.h>;
#include <SigFox.h>;
#include <ArduinoLowPower.h>;

#define ONE_WIRE_BUS 1
#define ONE_WIRE_BUS2 0

OneWire oneWire(ONE_WIRE_BUS);
OneWire oneWire2(ONE_WIRE_BUS2);

DallasTemperature sensors(&oneWire);
DallasTemperature sensors2(&oneWire2);


#define DHTPIN 2     // what pin we're connected to
#define DHTTYPE DHT22   // DHT 22  (AM2302)
DHT dht(DHTPIN, DHTTYPE); //// Initialize DHT sensor for normal 16mhz Arduino

 float Celcius=0;
 float Fahrenheit=0;
 float Celcius2=0;
 float Fahrenheit2=0;
 int chk;
 float hum;  //Stores humidity value
 float temp; //Stores temperature value

 int sensorPin = 2;
 float bodemtussenstap = 0;
 float bodemvochtigheid = 0;
void setup(void)
{
  
  Serial.begin(9600);
  while (!Serial) {};
  if (!SigFox.begin()) {
    Serial.println("Shield error or not present!");

    return;
  }
  sensors.begin();
  sensors2.begin();
  dht.begin();
}

void loop(void)
{ 
  hum = dht.readHumidity();
  temp= dht.readTemperature();
  bodemtussenstap = analogRead(sensorPin);
  bodemvochtigheid = (bodemtussenstap / 1023)*100;
  sensors.requestTemperatures(); 
  sensors2.requestTemperatures(); 
  Celcius=sensors.getTempCByIndex(0);
  Fahrenheit=sensors.toFahrenheit(Celcius);
  Celcius2=sensors2.getTempCByIndex(0);
  Fahrenheit2=sensors2.toFahrenheit(Celcius);
  //Temperatuur sensor1
  Serial.print("Temperatuur: ");
  Serial.print(Celcius);
  Serial.print("째C ");
  Serial.print(Fahrenheit);
  Serial.println("F  ");
  //Temperatuur sensor2 bodem
  Serial.print("Bodemtemperatuur: ");
  Serial.print(Celcius2);
  Serial.print("째C ");
  Serial.print(Fahrenheit2);
  Serial.println("F  ");
  //bodemvochtigheidsensor
  Serial.print("Bodemvochtigheid: " );
  Serial.print(bodemvochtigheid); 
  Serial.println("%" );
  //dht22 sensor
  Serial.print("Vochtigheid: ");
  Serial.print(hum);
  Serial.println("%");
  Serial.print("Temp: ");
  Serial.print(temp);
  Serial.println("째C");
  Serial.print("McuTemperatuur: ");
  Serial.print(SigFox.internalTemperature());
  Serial.println("째C");
  Serial.println(" -------------------------------------------------------------------------------------------------" );
  delay(1000);
}
